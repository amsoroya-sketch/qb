```javascript
/**
 * Generates contextual learning links for any piece of content
 * Based on user's level and current context
 */
function generateContextualLinks(context) {
  const links = [];

  // Context can be: word, lesson, exercise, verse
  const { type, data, userLevel } = context;

  if (type === 'word') {
    // User is viewing a word analysis
    const word = data;

    // Link to grammar lessons about this word type
    if (word.word_type === 'noun') {
      links.push({
        type: 'lesson',
        title: 'Learn about Nouns',
        url: '/learn/track-a/level-2/lesson-2-2-1',
        icon: 'ğŸ“š',
        relevance: 1.0,  // Highly relevant
        estimated_time: '12 min'
      });
    }

    // Link to case explanation if user is intermediate+
    if (userLevel >= 'intermediate' && word.analysis.pos_detailed.case) {
      const case_type = word.analysis.pos_detailed.case;
      links.push({
        type: 'lesson',
        title: `Learn about ${case_type} case`,
        url: `/learn/track-a/level-4/case-${case_type}`,
        icon: 'ğŸ“–',
        relevance: 0.9,
        estimated_time: '15 min'
      });
    }

    // Link to POS library
    links.push({
      type: 'pos_library',
      title: 'Quick Reference: Noun',
      url: '/pos-library/noun',
      icon: 'ğŸ“•',
      relevance: 0.8,
      estimated_time: '2 min'
    });

    // Link to root exploration
    if (word.analysis.morphology.root) {
      links.push({
        type: 'root',
        title: `Explore root ${word.analysis.morphology.root.arabic}`,
        url: `/roots/${word.analysis.morphology.root.arabic}`,
        icon: 'ğŸŒ±',
        relevance: 0.7,
        estimated_time: '5 min'
      });
    }
  }

  // Sort by relevance
  return links.sort((a, b) => b.relevance - a.relevance);
}
```
